  6 /**
  7  * 校验函数，校验成功返回true，失败返回false并显示提示信息
  8  * @param tipSelector 提示信息选择器，默认为#tip
  9  * @param dom dom对象（可选，默认为当前document，类型：jQuery选择器|jQuery对象|dom对象）
 10  * @returns {Boolean} 校验成功返回true，失败返回false
 11  */
 12 function doValidate(tipSelector, dom){
 13 	var message = "";
 14 	var result = true;
 15 	var domObj = null;
 16 	
 17 	if("string" == typeof(dom)){
 18 		domObj = jQuery(dom);
 19 	}else if("object" == typeof(dom)){
 20 		domObj = dom;
 21 	}else{
 22 		domObj = document;
 23 	}
 24 	
 25 	//验证文本、文本框、下拉列表
 26 	var requiredElements = jQuery(":text[class*='[required]']", domObj)
 27 	.add(":password[class*='[required]']", domObj)
 28 	.add("textarea[class*='[required]']", domObj)
 29 	.add("select[class*='[required]']", domObj);
 30 	requiredElements.each(function(){
 31 		var tag = this;
 32 		var tagName = tag.tagName.toUpperCase();
 33 		var value;
 34 		if("INPUT" == tagName){
 35 			var tagType = tag.type.toUpperCase();
 36 			if("TEXT" == tagType || "PASSWORD" == tagType){
 37 				value = jQuery.trim(tag.value);
 38 				if(!value){//未输入文本
 39 					message = tag.title;
 40 					result = false;
 41 				}
 42 			}			
 43 		}else if("TEXTAREA" == tagName){
 44 			value = jQuery.trim(tag.value);
 45 			if(!value){//未输入文本
 46 				message = tag.title;
 47 				result = false;
 48 			}
 49 		}else if(tagName == "SELECT"){
 50 			value = tag.value;
 51 			if( value == "" || (!isNaN(value) && parseInt(value) <= 0)){//select值为0 或小于0的则认为是未选择
 52 				message = tag.title;
 53 				result = false;
 54 			}
 55 		}
 56 		if(!result){
 57 			return false;
 58 		}
 59 	});
 60 	
 61 	//验证单选按钮
 62 	if(result){
 63 		var requiredRadios = jQuery(":radio[class*='[required]']", domObj);
 64 		var radioNames = new Object();
 65 		requiredRadios.each(function(){
 66 			var radio = this;
 67 			radioNames[radio.name] = radio.title ? radio.title : "";//记录radio的name属性 和title属性
 68 		});
 69 		//循环去验证每组radio
 70 		for(p in radioNames){
 71 			if(jQuery(":radio[name='" + p + "'][checked=true]", domObj).length == 0){//无选中的radio
 72 				result = false;
 73 				message = radioNames[p];
 74 				break;
 75 			};
 76 		}
 77 	}
 78 	
 79 	if(!result){
 80 		//top.CommonUI.alert(message, tipSelector);
 81 		alert(message);
 82 	}
 83 	return result;
 84 }

103 /**
104  * 数字检查
105  * @param str 要检查的字符串
106  * @param allowNegative 是否允许负数（默认为false，允许负数的情况下，也允许正数前有“+”号）
107  * @param fractionDigits 最多小数位数，整数
108  * @returns 满足条件返回true，否则返回false
109  */
110 function numberCheck(str, allowNegative, fractionDigits){
111 	var result = false;
112 	var regexStr = "";
113 	var regex;
114 	var _allowNegative = allowNegative ? allowNegative : false;
115 	var _fraction = "";
116 	var symbols = "";
117 	if(_allowNegative){
118 		symbols = "[+-]{0,1}"; 
119 	}else{
120 		if(null == fractionDigits){
121 			regexStr = "/^" + symbols + "\\d+\\.\\d+$|^" + symbols + "\\d+$/";
122 		}else if(0 == fractionDigits){
123 			regexStr = "/^" + symbols + "\\d+$/";
124 		}else if(fractionDigits > 0){
125 			regexStr = "/^" + symbols + "\\d+\\.\\d{1," + fractionDigits + "}$|^" + symbols + "\\d+$/";
126 		}
127 	}
128 	
129 	regex = eval(regexStr);	
130 	if(str){
131 		result = regex.test(str);
132 	}
133 	return result;
134 }

136 /**
137  * 只允许输入数字的事件验证
138  * @param inputObj 文本框对象
139  * @param errMsg 错误提示消息
140  * @param allowNegative  是否允许负数
141  * @param fractionDigits  小数位数，整数
142  * @returns
143  */
144 function onlyNumberEvent(event, allowNegative, fractionDigits){
145 	var result = true;
146 	var keyChar = "";
147 	var eventInfo = EventInfo.getEventInfo(event);
148 	var keyCode = eventInfo.keyCode;
149 	var inputObj = eventInfo.srcObj;
150 	
151 	if( 46 == keyCode ||//delete
152 		45 == keyCode ||//insert
153 		35 == keyCode ||//end
154 		36 == keyCode ||//home
155 		37 == keyCode ||//LeftArrow
156 		39 == keyCode ||//RightArrow
157 		8 == keyCode){//BackSpace
158 		
159 	}else{
160 		keyChar = String.fromCharCode(keyCode);
161 		if(inputObj){
162 			var result = numberCheck(inputObj.value + keyChar, allowNegative, fractionDigits);
163 			if(!result){
164 				result = false;
165 			}
166 		}
167 	}
168 	return result;
169 }

171 /**
172  * 检查URL格式是否正确
173  * @param str_url url字符串
174  * @returns 是URL返回true，否则返回false
175  */
176 function urlCheck(str_url){
177 	var strRegex = "^((https|http|ftp|rtsp|mms)?://)"
178 	+ "?(([0-9a-z_!~*'().&=+$%-]+: )?[0-9a-z_!~*'().&=+$%-]+@)?" //ftp的user@
179 	+ "(([0-9]{1,3}.){3}[0-9]{1,3}" // IP形式的URL- 199.194.52.184
180 	+ "|" // 允许IP和DOMAIN（域名）
181 	+ "([0-9a-z_!~*'()-]+.)*" // 域名- www.
182 	+ "([0-9a-z][0-9a-z-]{0,61})?[0-9a-z]." // 二级域名
183 	+ "[a-z]{2,6})" // first level domain- .com or .museum
184 	+ "(:[0-9]{1,4})?" // 端口- :80
185 	+ "((/?)|" // a slash isn't required if there is no file name
186 	+ "(/[0-9a-z_!~*'().;?:@&=+$,%#-]+)+/?)$";
187 	
188 	var re=new RegExp(strRegex, "i");  
189 	return !/^\w+$/.test(str_url) && //排除纯数字和字母组合
190 			re.test(str_url);
191 }

218  */
219 function mobilePhoneCheck(str){
220 	var flag = false;
221 	var unicom = [130, 131, 132, 145, 155, 156, 185, 186];

228 			flag = true;
229 		}
230 	}
231

235 /**
236  * 获取主域名
237  * @param url
238  * @returns
239  */
240 function getDomain(url){
241 	if(!url){
242 		return null;
243 	}
244 	var temp = url.toLowerCase().replace("http://", "");
245 	var firstPathSeperatorIndex = temp.indexOf("/");
246 	return (firstPathSeperatorIndex > 0 ? temp.substring(0, firstPathSeperatorIndex) : temp);
247 }

249 /**
250  * 验证给定的URL与给定的主域是否一致
251  * @param domain 主域名
252  * @param url 要验证的URL
253  * @returns {Boolean}
254  */
255 function validateDomain(domain, url){
256 	var result = false;
257 	if(domain && url){
258 		var realDomain = getDomain(domain);
259 		var urlDomain = getDomain(url);
260 		result = (realDomain == urlDomain || ("www." + realDomain) == urlDomain);
261 	}
262 	return result;
263 }
