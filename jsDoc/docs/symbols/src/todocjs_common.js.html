 77 /**
 78  * 日期加减
 79  * @param date 要加减的日期 
 80  * @param field 要加减的域（年、月、日、时、分、秒，参考DateField常量）
 81  * @param value 域的值
 82  */
 83 function dateAdd(date, field, value){
 84 	if(date && field && value && !isNaN(value)){
 85 		var val;		
 86 		if("number" == typeof(value)){
 87 			val = value;
 88 		}else if("string" == typeof(value)){
 89 			val = parseInt(value);			
 90 		}else{
 91 			throw "dateAdd: param value should be an integer";
 92 		}
 93 		if(DateField.prototype.YEAR == field){
 94 			date.setFullYear(date.getFullYear() + val);
 95 		}else if(DateField.prototype.MONTH == field){
 96 			date.setMonth(date.getMonth() + val);
 97 		}else if(DateField.prototype.DATE == field){
 98 			date.setDate(date.getDate() + val);
 99 		}else if(DateField.prototype.HOUR == field){
100 			date.setHours(date.getHours() + val);
101 		}else if(DateField.prototype.MINUTE == field){
102 			date.setMinutes(date.getMinutes() + val);
103 		}else if(DateField.prototype.SECOND == field){
104 			date.setSeconds(date.getSeconds() + val);
105 		}
106 	}	
107 }

109 /**
110  * 处理ajax错误
111  * @param error 错误信息
112  * @param callback 此回调函数用于传入自己的错误处理逻辑，当session超时时，此回调函数不会被执行
113  * @return 如果是超时则返回true，否则返回false
114  * @author 胡海
115  * @since 2011-11-23
116  */
117 function handleAjaxError(error, callback){
118 	var SESSION_TIME_OUT_AJAX_ERR_MSG = "TIME_OUT";//登录超时
119 	if(error == SESSION_TIME_OUT_AJAX_ERR_MSG){
120 		alert("由于您长时间未操作，您与服务器的连接已断开，请重新登录");
121 		window.top.location.href = $WEB_ROOT_PATH;//跳转到登录页
122 		return true;
123 	}
124 	if(callback){
125 		callback();
126 	}
127 	return false;

130 /**
131  * ajax请求（忽略name为空的标签，默认忽略disabled的标签的数据）
132  * @param url 提交地址
133  * @param container 数据所在的dom容器，可以是：dom元素的id | jQuery选择器 | dom对象 | jQuery对象
134  * @param successHandler 成功处理事件 （可选，一般都要设置）
135  * @param errorHandler 出错处理事件 （可选，一般都要设置）
136  * @param options jQuery ajax选项，传入后将覆盖默认设置（可选，此参数可传入自定义属性，
137  * skipDisabled：是否忽略disabled的标签，默认为true；isValueData：第6个参数传入的dataSource是否是数据值对象，默认为true，否则为获取数据的表达式对象）
138  * @param dataSource 提交数据源（支持按优先级进行数据覆盖，优先级：dataSource > formData（container中获取的数据） > options.data）
139  * @param preproccess 预处理数据的函数（提交之前对数据进行预处理）
140  */
141 function ajaxRequest(url, container, successHandler, errorHandler, options, dataSource, preproccess){
142 	var _options = {dataType: "text", type: "POST", data:{}, skipDisabled: true, isValueData: true};
143 	if(url){
144 		_options.url = url;
145 	}
146 	if(successHandler){
147 		_options.success = successHandler;
148 	}
149 	if(errorHandler){
150 		_options.error = function(xhr, status, error){
151 			handleAjaxError(error, errorHandler);
152 		};
153 	}
154 	
155 	if(options){
156 		jQuery.extend(_options, options);		
157 	}
158 	
159 	var data1 = getFormData(container, _options.skipDisabled);
160 	var data2 = collectData(dataSource, _options.isValueData);
161 	
162 	if(data1){
163 		jQuery.extend(_options.data, data1);
164 	}
165 	if(data2){
166 		jQuery.extend(_options.data, data2);
167 	}
168 	
169 	if(preproccess){
170 		_options.data = preproccess(_options.data);
171 	}
172 	jQuery.ajax(_options);
173 }

176  * 为select加载数据
177  * @param select select标签标识，可以是：dom元素的id | jQuery选择器 | dom对象 | jQuery对象
178  * @param url 加载数据的URL
179  * @param name 数据中作为option显示项的属性名
180  * @param value 数据中作为option值的属性名
181  * @param hasDefault 是否有默认选项
182  * @param dataKey 数据的key（当数据在一Map中时使用此参数）
183  */
184 function loadSelect(select, url, optionName, optionValue, hasDefault, dataKey){
185 	var sel = getJqueryDomElement(select);
186 	if(sel.length == 1){
187 		ajaxRequest(url, null, function(data){
188 			if(false == hasDefault){
189 				sel[0].options.length = 0;
190 			}else{
191 				sel[0].options.length = 1;
192 			}	
193 			if(data && ("null" != data) && ("[]" != data)){
194 				var list = eval(data);
195 				if(dataKey){
196 					list = data[dataKey];
197 				}
198 				for(var i = 0; i < list.length; i++){
199 					sel[0].options[sel[0].options.length] = new Option(list[i][optionName], list[i][optionValue]);
200 				}
201 			}
202 		},
203 		function(xhr, status, error){
204 			handleAjaxError(error, function(){
205 				alert("加载下拉列表数据失败");
206 			});
207 		});
208 	}else{
209 		//传入的select表达式有误
210 	}

351  * @returns
352  */

354 	var result = "";

356 		var _obj;
357 		if("string" == typeof(obj)){
358 			_obj = document.getElementById(obj);

364 			var matches = _obj.className.match(regex);
365 			if(matches){
366 				result = matches[1];
367 			}
368 		}
369 	}
370 	return result;

602 /**
603  * 遍历dom元素
604  * @param container dom容器，可以是：dom元素的id | jQuery选择器 | dom对象 | jQuery对象
605  * @param handlers 类型：DomHandler，遍历时的处理函数及参数
606  * @param skipDisabled 是否跳过disabled元素
607  */
608 function loopDom(container, handlers, skipDisabled){
609 	var dom = document;
610 	if(container){
611 		dom = getJqueryDomElement(container);
612 	}
613 	var elements = jQuery("input:text", dom)
614 	.add("input:password", dom)
615 	.add("input:radio", dom)
616 	.add("input:checkbox", dom)
617 	.add("input:hidden", dom)
618 	.add("textarea", dom)
619 	.add("select", dom);
620 	
621 	var _skipDisabled = (skipDisabled != null) ? skipDisabled : true;
622 	
623 	if(elements && elements.length > 0 && handlers){
624 		elements.each(function(){
625 			var tag = this;
626 			var tagName = tag.tagName ? tag.tagName.toUpperCase() : null;
627 			if(tagName){
628 				if(!(_skipDisabled && tag.disabled)){
629 					handlers.defaultHandlerArgs.tag = tag;
630 					if("INPUT" == tagName){
631 						var tagType = tag.type ? tag.type.toUpperCase() : null;
632 						if(tagType){
633 							if("TEXT" == tagType){
634 								jQuery.extend(handlers.textareaHandlerArgs, {tag: tag});
635 								handlers.textareaHandler(handlers.textareaHandlerArgs);
636 							}else if("PASSWORD" == tagType){
637 								jQuery.extend(handlers.passwordHandlerArgs, {tag: tag});
638 								handlers.passwordHandler(handlers.passwordHandlerArgs);
639 							}else if("RADIO" == tagType){
640 								jQuery.extend(handlers.radioHandlerArgs, {tag: tag});
641 								handlers.radioHandler(handlers.radioHandlerArgs);
642 							}else if("CHECKBOX" == tagType){
643 								jQuery.extend(handlers.checkboxHandlerArgs, {tag: tag});
644 								handlers.checkboxHandler(handlers.checkboxHandlerArgs);
645 							}else if("HIDDEN" == tagType){
646 								jQuery.extend(handlers.hiddenHandlerArgs, {tag: tag});
647 								handlers.hiddenHandler(handlers.hiddenHandlerArgs);
648 							}
649 						}
650 					}else if("TEXTAREA" == tagName){
651 						jQuery.extend(handlers.textareaHandlerArgs, {tag: tag});
652 						handlers.textareaHandler(handlers.textareaHandlerArgs);
653 					}else if("SELECT" == tagName){
654 						jQuery.extend(handlers.selectHandlerArgs, {tag: tag});
655 						handlers.selectHandler(handlers.selectHandlerArgs);
656 					}
657 				}
658 			}						
659 		});
660 	}
661 }

667  */
668 function getFormData(container, skipDisabled){
669 	if(null == container){

671 	}

673 	var dataIndex = {};
674 	var domObj = getJqueryDomElement(container);

676 	handlers.defaultHandler = function(args){
677 		if(args.tag.name){
678 			data[args.tag.name]= args.tag.value;
679 		}

782 /**
783  * 自动补全文本框中字符对
784  * @param element 文本框
785  * @param event 事件
786  * @param charMap 字符对map
787  * @param callback 回调函数，可用于设置光标位置，函数参数(element, index, leftStr, rightStr, codeChar, charToAdd)
788  */
789 function pairComplete(element, event, charMap, callback){
790 	var index = getcaretPos(element);//光标位置
791 	var eventInfo = EventInfo.getEventInfo(event);
792 	var codeChar = String.fromCharCode(eventInfo.keyCode);//当前字符
793 	var charToAdd = charMap[codeChar];//将要补全的字符
794 	if(charToAdd){
795 		var leftStr = element.value.substring(0, index);//未补全时光标左侧字符串
796 		var rightStr = element.value.substring(index);//未补全时光标右侧字符串
797 		element.value = leftStr + codeChar + charToAdd + rightStr;
798 		callback(element, index, leftStr, rightStr, codeChar, charToAdd);//执行回调函数
799 		if(event.preventDefault){
800 			event.preventDefault();
801 		}else{
802 			event.returnValue = false;
803 		}
804 	}
805 }

807 /**
808  * 获取日期整形字符串
809  * @param date1 日期对象或日期字符串（支持yyyy-mm-dd、yyyy/mm/dd、yyyymmdd）
810  * @returns {Date}
811  */
812 function getDateIntStr(date1){
813 	var d1 = null;
814 	if(date1 instanceof Date){
815 		var d1 = new Date();d1.getDate()
816 		var month = date1.getMonth() + 1;
817 		var date = date1.getDate();
818 		d1 = "" + date1.getFullYear() + (month > 9 ? month : '0' + month) +  (date > 9 ? date : '0' + date);
819 	}else if("string" == typeof(date1)){
820 		if(/^\d{4}-\d{2}-\d{2}$/.test(date1)){
821 			d1 = date1.replace(/-/g, "");
822 		}else if(/^\d{4}\/\d{2}\/\d{2}$/.test(date1)){
823 			d1 = date1.replace(/\//g, "");
824 		}else if(/^\d{8}$/.test(date1)){
825 			d1 = date1;
826 		}
827 	}
828 	return d1;
829 }

831 /**
832  * 比较两个日期
833  * @param date1
834  * @param date2
835  * @returns {Number} -2:传入了为null的参数，-1:date1 < date2，0:date1 = date2，1:date1 > date2
836  */
837 function dateCompare(date1, date2){
838 	var result = -2;
839 	if(date1 && date2){
840 		var d1 = getDateIntStr(date1);
841 		var d2 = getDateIntStr(date2);
842 		if(null != d1 && null != d2){
843 			var d1Int = parseInt(d1);
844 			var d2Int = parseInt(d2);
845 			
846 			if(d1Int > d2Int){
847 				result = 1;
848 			}else if(d1Int == d2Int){
849 				result = 0;
850 			}else{
851 				result = -1;
852 			}
853 		}
854 	}
855 	return result;
856 }

858 /**
859  * 添加cookie变量
860  * @param name 变量名
861  * @param value 变量值
862  * @param expires 过期时间（天）
863  * @param path 指定可访问cookie的路径
864  */
865 function addCookie(name, value, expires, path){
866 	var str = name + "=" + escape(value);
867 	if(expires){
868 		var date = new Date();
869 		date.setTime(date.getTime() + expires * 24 * 3600 * 1000);//expires单位为天
870 		str += ";expires=" + date.toGMTString();
871 	}
872 	if(path){
873 		str += ";path=" + path;//指定可访问cookie的路径
874 	}
875 	str += ";domain=" + document.location.hostname;//指定可访问cookie的域
876 	document.cookie = str;
877 }

882  * @returns
883  */
884 function getCookie(cookie_name){
885 	var value = null;
886 	var allcookies = document.cookie;
887 	var cookie_pos = allcookies.indexOf(cookie_name);
888 	
889 	if (cookie_pos != -1){		
890 		cookie_pos += cookie_name.length + 1;
891 		var cookie_end = allcookies.indexOf(";", cookie_pos);
892 		if (cookie_end == -1)
893 		{
894 			cookie_end = allcookies.length;
895 		}
896 		value = unescape(allcookies.substring(cookie_pos, cookie_end));
897 	}
898 
899 	return value;
